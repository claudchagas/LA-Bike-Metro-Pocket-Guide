\subsubsection*{Route 1: Alhambra--Pasadena Loop}

Alhambra $\rightarrow$ South Pasadena $\rightarrow$ Old Pasadena (Colorado Blvd) $\rightarrow$ Huntington Library area $\rightarrow$ San Marino $\rightarrow$ Return\\[3pt]

\subsubsection*{Route 2: San Gabriel Valley Cultural Tour}

\begin{routebox}
\textbf{Distance:} 12 miles | \textbf{Time:} 2--3 hrs | \textbf{Difficulty:} Easy\\[3pt]
Alhambra $\rightarrow$ San Gabriel Mission $\rightarrow$ Temple City $\rightarrow$ Arcadia (REI) $\rightarrow$ Santa Anita Park $\rightarrow$ Return\\[3pt]
\textbf{Highlights:} Historic mission, parks, shopping
\end{routebox}

\subsubsection*{Route 3: LA River Trail}

\begin{routebox}
\textbf{Distance:} 15 miles one-way | \textbf{Time:} Half day | \textbf{Difficulty:} Moderate\\[3pt]
Alhambra $\rightarrow$ LA River bike path $\rightarrow$ Griffith Park $\rightarrow$ (optional: Burbank)\\[3pt]
\textbf{Highlights:} Dedicated bike path, nature, Griffith Observatory
\end{routebox}

\subsubsection*{Route 4: Rose Bowl \& Arroyo Seco}

\begin{routebox}
\textbf{Distance:} 8 miles | \textbf{Time:} 2 hrs | \textbf{Difficulty:} Moderate (hills)\\[3pt]
Metro to Memorial Park $\rightarrow$ Rose Bowl $\rightarrow$ Arroyo Seco trails $\rightarrow$ Lower Arroyo Park\\[3pt]
\textbf{Highlights:} Stadium, natural trails, scenic views
\end{routebox}

\subsection*{\faWrench\ Bike Safety \& Maintenance}

\begin{tipbox}[title=Safety First:]
\begin{small}
\begin{itemize}[leftmargin=0.2in,itemsep=1pt]
\item Always wear helmet (available at REI)
\item Use front white light + rear red light (required at night)
\item Stay in bike lanes when available
\item Watch for car doors opening (ride 3ft from parked cars)
\item Follow traffic signals -- bikes = vehicles
\end{itemize}

\textbf{Security:}
\begin{itemize}[leftmargin=0.2in,itemsep=1pt]
\item ALWAYS lock bike, even for 1 minute
\item Use U-lock through frame + rear wheel
\item Cable lock for front wheel
\item Lock to fixed objects in visible areas
\item Take lights and computer with you
\end{itemize}
\end{small}
\end{tipbox}
